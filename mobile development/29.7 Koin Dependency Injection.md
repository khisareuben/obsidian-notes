```kotlin
class MyViewModel(private val repository: Repository) : ViewModel() {
    private val _data = MutableLiveData<String>()
```

## 1. Add Dependencies

In your `build.gradle`:



```kotlin
dependencies {
    implementation("io.insert-koin:koin-android:3.5.3")
    implementation("io.insert-koin:koin-androidx-compose:3.5.3")
    implementation("com.squareup.retrofit2:retrofit:2.11.0")
    implementation("com.squareup.retrofit2:converter-gson:2.11.0")
    implementation("com.squareup.okhttp3:okhttp:4.12.0")
}
```

## ðŸ§© 2. Create Your API Interface



```kotlin
interface MyApi {
    @GET("data")
    suspend fun getData(): String

    companion object {
        const val BASE_URL = "https://api.example.com/"
    }
}
```

## ðŸ§ª 3. Define Your Koin Module



```kotlin
val networkModule = module {
    single {
        Retrofit.Builder()
            .baseUrl(MyApi.BASE_URL)
            .addConverterFactory(GsonConverterFactory.create())
            .client(OkHttpClient.Builder().build())
            .build()
            .create(MyApi::class.java)
    }

    single<Repository> { RepoImpl(get()) }
}
```

## ðŸ§  4. Create Your Repository



```kotlin
interface Repository {
    suspend fun getData(): String
}

class RepoImpl(private val api: MyApi) : Repository {
    override suspend fun getData(): String = api.getData()
}
```

## ðŸ§° 5. Initialize Koin in Your Application Class



```kotlin
class MyApp : Application() {
    override fun onCreate() {
        super.onCreate()
        startKoin {
            androidContext(this@MyApp)
            modules(networkModule)
        }
    }
}
```
